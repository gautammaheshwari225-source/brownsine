<script>
    // ---------- PRODUCT DATA ----------
    const products = [
        { id: 1, name: "Monochrome Silk Blazer", price: 349, category: "Fashion", isBestSeller: true, imgIcon: "fas fa-tshirt", featured: true },
        { id: 2, name: "GMT Wrist Chronograph", price: 599, category: "Accessories", isBestSeller: true, imgIcon: "fas fa-clock", featured: true },
        { id: 3, name: "Aura Leather Tote", price: 279, category: "Lifestyle", isBestSeller: false, imgIcon: "fas fa-bag-shopping", featured: true },
        { id: 4, name: "Minimalist Sneakers", price: 189, category: "Trending", isBestSeller: true, imgIcon: "fas fa-shoe-prints", featured: true },
        { id: 5, name: "Gold Frame Sunglasses", price: 229, category: "Accessories", isBestSeller: false, imgIcon: "fas fa-glasses", featured: false },
        { id: 6, name: "Cashmere Scarf", price: 159, category: "Fashion", isBestSeller: true, imgIcon: "fas fa-scarf", featured: false },
        { id: 7, name: "Aesthetic Candle Set", price: 89, category: "Lifestyle", isBestSeller: false, imgIcon: "fas fa-fire", featured: false },
        { id: 8, name: "Tech Organizer", price: 99, category: "Trending", isBestSeller: false, imgIcon: "fas fa-laptop", featured: false },
    ];

    let cart = []; // {id, name, price, quantity}
    let wishlistIds = new Set();

    // DOM utils
    function renderFeatured() {
        const featuredProds = products.filter(p => p.featured);
        const grid = document.getElementById('featuredGrid');
        grid.innerHTML = featuredProds.map(p => productCardHTML(p)).join('');
        attachProductEvents();
    }

    function renderBestSellers() {
        const best = products.filter(p => p.isBestSeller);
        const grid = document.getElementById('bestsellersGrid');
        grid.innerHTML = best.map(p => productCardHTML(p)).join('');
        attachProductEvents();
    }

    function renderCategories() {
        const cats = ["Fashion", "Accessories", "Lifestyle", "Trending"];
        const catGrid = document.getElementById('categoriesGrid');
        catGrid.innerHTML = cats.map(cat => `
            <div class="category-card" data-category="${cat}">
                <i class="fas ${cat === 'Fashion' ? 'fa-vest' : cat === 'Accessories' ? 'fa-gem' : cat === 'Lifestyle' ? 'fa-couch' : 'fa-chart-line'}"></i>
                <h3>${cat}</h3>
                <p>Shop collection</p>
            </div>
        `).join('');
        document.querySelectorAll('.category-card').forEach(card => {
            card.addEventListener('click', (e) => {
                const cat = card.dataset.category;
                const filtered = products.filter(p => p.category === cat);
                alert(`✨ Explore ${cat} collection: ${filtered.map(p=>p.name).join(', ')}`);
            });
        });
    }

    function productCardHTML(p) {
        const isWished = wishlistIds.has(p.id);
        return `
            <div class="product-card" data-id="${p.id}">
                <div class="product-img"><i class="${p.imgIcon}" style="font-size:3rem;"></i></div>
                <div class="product-info">
                    <div class="product-title">${p.name}</div>
                    <div class="product-price">$${p.price}</div>
                    <div class="card-actions">
                        <button class="btn-add" data-id="${p.id}">Add to cart</button>
                        <button class="btn-wish" data-id="${p.id}"><i class="${isWished ? 'fas' : 'far'} fa-heart"></i> Wish</button>
                    </div>
                </div>
            </div>
        `;
    }

    function attachProductEvents() {
        document.querySelectorAll('.btn-add').forEach(btn => {
            btn.removeEventListener('click', addHandler);
            btn.addEventListener('click', addHandler);
        });
        document.querySelectorAll('.btn-wish').forEach(btn => {
            btn.removeEventListener('click', wishHandler);
            btn.addEventListener('click', wishHandler);
        });
    }

    function addHandler(e) {
        const id = parseInt(e.currentTarget.dataset.id);
        const product = products.find(p => p.id === id);
        if (product) {
            const existing = cart.find(item => item.id === id);
            if (existing) existing.quantity++;
            else cart.push({ ...product, quantity: 1 });
            updateCartUI();
            showToast("Added to bag ✨");
        }
    }

    function wishHandler(e) {
        const id = parseInt(e.currentTarget.dataset.id);
        if (wishlistIds.has(id)) wishlistIds.delete(id);
        else wishlistIds.add(id);
        renderFeatured();
        renderBestSellers();
        showToast(wishlistIds.has(id) ? "Added to wishlist ❤️" : "Removed from wishlist");
    }

    function updateCartUI() {
        const count = cart.reduce((sum, i) => sum + i.quantity, 0);
        document.getElementById('cartCount').innerText = count;
        const cartDiv = document.getElementById('cartItemsList');
        if (!cartDiv) return;
        if (cart.length === 0) cartDiv.innerHTML = '<p>Your bag is empty, elegant.</p>';
        else {
            cartDiv.innerHTML = cart.map(item => `
                <div class="cart-item">
                    <span>${item.name} x${item.quantity}</span>
                    <span>$${item.price * item.quantity}</span>
                    <button class="remove-item" data-id="${item.id}" style="background:none; color:var(--gold); border:none; cursor:pointer;">🗑️</button>
                </div>
            `).join('');
            document.querySelectorAll('.remove-item').forEach(btn => {
                btn.addEventListener('click', (e) => {
                    const id = parseInt(btn.dataset.id);
                    cart = cart.filter(item => item.id !== id);
                    updateCartUI();
                });
            });
        }
        const total = cart.reduce((sum, i) => sum + (i.price * i.quantity), 0);
        document.getElementById('cartTotal').
